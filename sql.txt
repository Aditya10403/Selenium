// Write an SQL query to show the second highest salary from a table using sub-query.

	SELECT MAX(SALARY) FROM WORKER WHERE SALARY NOT IN (SELECT MAX(SALARY) FROM WORKER)

// Write an SQL query to fetch the departments that have less than 4 people in it.
	
	SELECT DEPARTMENT, COUNT(DEPARTMENT) AS DEPTCOUNT FROM WORKER GROUP BY DEPARTMENT HAVING DEPTCOUNT < 4;
	

// Write an SQL query to print the name of employees having the highest salary in each department.

	SELECT W.DEPARTMENT, W.FIRSTNAME, W.SALARY 
	FROM (SELECT MAX(SALARY) AS MAXSAL, DEPARTMENT FROM WORKER GROUP DEPARTMENT) TEMP
	INNER JOIN WORKER W ON TEMP.DEPARTMENT = W.DEPARTMENT AND TEMP.MAXSAL = W.SALARY;


// Write an SQL query to fetch three max/min salaries from a table using co-related subquery and limit.

	SELECT DISTINCT SALARY FROM WORKER W1
	WHERE 3 >= (SELECT COUNT(DISTICT SALARY) FROM WORKER W2 WHERE W2.SALARY >= W1.SALARY) ORDER BY W1.SALARY DESC;
	OR 
	SELECT DISTICT SALARY FROM WORKER ORDER BY SALARY LIMIT 3;


// Write an SQL query to determine the 5th highest salary without using LIMIT keyword.

	SELECT SALARY FROM WORKER W1
	WHERE N-1 = (SELECT COUNT(DISTINCT SALARY) FROM WORKER W2 WHERE W2.SALARY >= W1.SALARY);
	OR
	SELECT DISTINCT SALARY FROM WORKER ORDER BY SALARY DESC LIMIT 4, 1;

// weather -> id, temp, date 
	SELECT W1.ID, W1.TEMP FROM WEATHER W1
	INNER JOIN WEATHER W2 ON DATEDIFF(W1.DATE, W2.DATE) = 1
	WHERE W2.TEMP > W1.TEMP;

// EMPLOYEE WHO EARN MORE THAN MANAGER - ID, NAME, SALARY, MANAGERID

	SELECT E1.NAME, E1.SALARY FROM EMPLOYEE E1
	INNER JOIN EMPLOYEE E2 ON E1.ID = E2.ID WHERE E1.SALARY > E2.SALARY;
